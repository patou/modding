package com.sfeir.modding.client.showcase.view;

import java.util.Map;

import com.google.gwt.user.client.Command;
import com.google.gwt.user.client.Window;
import com.google.gwt.user.client.ui.FormPanel.SubmitEvent;
import com.sfeir.modding.client.app.FormActivity;
import com.sfeir.modding.client.component.forms.LabelTextArea;
import com.sfeir.modding.client.component.forms.LabelTextBox;
import com.sfeir.modding.client.component.menu.MenuItem;
import com.sfeir.modding.client.component.validator.RegexValidator;

public class ContactActivity extends FormActivity {
    public ContactActivity() {
        
    }

    @Override
    public void onCreateForm() {
        LabelTextBox elementName = new LabelTextBox("Your name", true);
        elementName.setName("name");
        addElement(elementName);
        LabelTextBox elementEmail = new LabelTextBox("Your email", true);
        elementEmail.addValidator(new RegexValidator("^([a-z0-9_.-]+)@([a-z0-9_.-]+)$","Please enter a correct email"));
        addElement(elementEmail);
        addElement(new LabelTextArea("Comment", true));
        addButton(new MenuItem("Cancel", new Command() {
            
            @Override
            public void execute() {
                finish();
            }
        }));
        addSubmitButton("Send");
    }
    
    @Override
    public void onSubmit(SubmitEvent event) {
        Map<String,Object> values = getValues();
        String name = (String) values.get("name");
        Window.alert("Thanks " + name + " for your message");
        finish();
    }
}
